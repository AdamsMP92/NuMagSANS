* Setup
Minimal imports and physical parameters.

#+begin_src python
import math
from pathlib import Path

from NuMagSANS import NuMagSANS
from NuMagSANS.SystemDesigner.Workflows import (
    write_spherical_replication_vectorfield_sweep,
)

base_dir = Path("hyperbolic_vortex_R20")
n_replications = 1024
R = 20.0
scattering_volume = n_replications * (4.0 / 3.0) * math.pi * (R * 1e-9) ** 3
#+end_src

* Generate Real-Space Data
One local sphere, one hyperbolic vortex field, one isotropic RotData file.

#+begin_src python
summary = write_spherical_replication_vectorfield_sweep(
    output_dir=base_dir,
    R=R,
    a=1.0,
    atomtype="Fe",
    n_replications=n_replications,
    field_parameter_cases={
        "field_type": "vortex",
        "profile_type": "hyperbolic",
        "xi_type": "cylindrical_xi",
        "kappa": 3.0,
        "turns": 1.0,
    },
    n_rotdata=1,
    beta_min=0.0,
    beta_max=math.pi,
    rotation_seed=1,
)
#+end_src

* Configure NuMagSANS
Use the workflow summary directly as config input.

#+begin_src python
sim = NuMagSANS()
config = base_dir / "NuMagSANSInput_temp.conf"
cfg = summary["config_hints"]

sim.write_config(
    config,
    **cfg,
    foldernameSANSData=str(base_dir / "NuMagSANS_Output"),
    SANSData_Output_Format="csv",
    Fourier_Approach="atomistic",
    Scattering_Volume_V=scattering_volume,
    enable_outputs=["SpinFlip_2D"],
)
#+end_src

* Run SpinFlip 2D
Only request CSV output for the 2D spin-flip cross section.

#+begin_src python
sim.run(config)
sim.config_clear(config)
#+end_src
