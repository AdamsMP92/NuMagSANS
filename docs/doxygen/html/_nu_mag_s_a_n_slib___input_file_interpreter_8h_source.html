148// zy-rotation matrix where alpha is the polar angle and beta is the azimuth angle
149void Compute_RotMat(float alpha, float beta, float* RotMat){
150
151 // ordering of the rotation matrix:
152 // RotMat = [RotMat[0], RotMat[3], RotMat[6]; ...
153 // RotMat[1], RotMat[4], RotMat[7]; ...
154 // RotMat[2], RotMat[5], RotMat[8]]
155
156
157 alpha = alpha * M_PI/180.0;
158 beta = beta * M_PI/180.0;
159
160 RotMat[0] = cos(alpha) * cos(beta);
161 RotMat[1] = cos(alpha) * sin(beta);
162 RotMat[2] = -sin(alpha);
163 RotMat[3] = -sin(beta);
164 RotMat[4] = cos(beta);
165 RotMat[5] = 0;
166 RotMat[6] = cos(beta) * sin(alpha);
167 RotMat[7] = sin(alpha) * sin(beta);
168 RotMat[8] = cos(alpha);
169
170 }

173string extract_value(const string &line){
174 size_t equality_symbol_index = line.find("=");
175 size_t semicolon_symbol_index = line.find(";");
176 size_t space_symbol_index = 0;
177 string string_to_be_interpreted = line.substr(equality_symbol_index+1, semicolon_symbol_index-equality_symbol_index-1);
178 while(string_to_be_interpreted.find(" ") <= string_to_be_interpreted.size()){
179 space_symbol_index = string_to_be_interpreted.find(" ");
180 string_to_be_interpreted.erase(space_symbol_index, 1);
181 }
182 return string_to_be_interpreted;
183}

221 // Interpreting User_Selection to integer array ##########################################
222void User_Selection_To_Int_Array(int *Idx, int Number_of_Comma, string my_string){
223
224 int Symbol_Idx[Number_of_Comma + 2];
225 string Symbol_Start = "{";
226 string Symbol_End = "}";
227 string Symbol_Comma = ",";
228 int Symbol_Counter = 0;
229 for(int i = 0; i < my_string.size(); i++){
230 if(my_string.at(i) == Symbol_Start.at(0) || my_string.at(i) == Symbol_End.at(0) || my_string.at(i) == Symbol_Comma.at(0)){
231 Symbol_Idx[Symbol_Counter] = i;
232 Symbol_Counter += 1;
233 }
234 }
235
236 string Sub_String;
237 int space_symbol_index;
238 for(int i = 0; i <Number_of_Comma + 1; i++){
239 Sub_String = my_string.substr(Symbol_Idx[i]+1, Symbol_Idx[i+1]-Symbol_Idx[i]-1);
240 while(Sub_String.find(" ") <= Sub_String.size()){
241 space_symbol_index = Sub_String.find(" ");
242 Sub_String.erase(space_symbol_index, 1);
243 }
244 Idx[i] = stoi(Sub_String);
245 }
246}

248// Find Number of Comma in string #########################################################
249void Number_Of_Comma_In_String(int *N, string my_string){
250 int String_Length = my_string.size();
251 int Comma_Counter = 0;
252 string Comma = ",";
253
254 for(int i = 0; i < String_Length; i++){
255 if(my_string.at(i) == Comma.at(0)){
256 Comma_Counter += 1;
257 }
258 }
259 *N = Comma_Counter;
260}
